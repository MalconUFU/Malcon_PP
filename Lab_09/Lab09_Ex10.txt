/*10. Crie um programa que contenha um array de inteiros contendo 5 elementos. Utilizando
apenas aritm√©tica de ponteiros, leia esse array do teclado e imprima o dobro de cada valor

#include <stdio.h>
#include <stdlib.h>
    
int main(){

    int vet[5], i;
    int *ponteiro = vet;

    for(i = 0; i < 5; i++){

        scanf("%d", ponteiro);
        ponteiro++;

    }

    ponteiro = vet;
    printf("\n");

    for (i = 0; i < 5; i++){

        printf("%d\n", *ponteiro * 2);
        ponteiro++;

    }

    return 0;
}
