/*Matrizes - 2. Declare uma matriz 5 Ã— 5. Preencha com 1 a diagonal principal e com 0 os demais elementos. 
Escreva ao final a matriz obtida*/

#include <stdio.h>

int main(){

    int A[5][5], k = 0;

    for (int i = 0; i < 5; i++)
    {

        for (int j = 0; j < 5; j++)
        {   

            if(k == j){

                A[i][j] = 1;
                j++;
                
            }

            A[i][j] = 0;

            }

            k++;
        
        }

        for (int i = 0; i < 5; i++)
    {
        for (int j = 0; j < 5; j++)
        {
            
            printf("%d", A[i][j]);

            }

            printf("\n");
        
        }

    return 0;   
}
