/* 2. Faça um programa que leia um vetor de 8 posições e, em seguida, leia também dois valores
X e Y quaisquer correspondentes a duas posições no vetor. Ao final seu programa deverá
escrever a soma dos valores encontrados nas respectivas posições X e Y.
*/

#include <stdio.h>
#include <stdlib.h>

int main()
{
    int A[8];
    int x, y, soma;

    printf("Digites os valores do vetor:\n");

    for(int i = 0; i < 8; i++){

        scanf("%d", &A[i]);
    }

    printf("\nDigite duas posicoes para somar os valores: ");

    do{
    scanf("%d", &x);

    if((x < 0) || (x > 7)){

        printf("primeiro valor invalido");
    }

    scanf("%d", &y);

    if((y < 0) || (y > 7)){

        printf("segundo valor invalido");
    }

    }while((x < 0) || (x > 7) || (y < 0) || (y > 7));

    soma = A[x] + A[y];

    printf("A soma sera: %d", soma);


    return 0;
}
